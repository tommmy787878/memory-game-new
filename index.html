<script>

const baseCards=["AS","AH","KD","KC","QS"];

const getImg=c=>`https://deckofcardsapi.com/static/img/${c}.png`;

const backImg="https://deckofcardsapi.com/static/img/back.png";


let board=document.getElementById("board");

let first=null, second=null, lock=false;

let turn=0, scores=[0,0], memory={}, deck=[];


/* スタート */

function startGame(){

document.getElementById("startScreen").style.display="none";

document.getElementById("game").style.display="block";

initGame();

}


/* リスタート */

function restartGame(){

let diff=document.getElementById("difficultyResult").value;

document.getElementById("difficulty").value=diff;


document.getElementById("result").style.display="none";

initGame();

}


/* 初期化 */

function initGame(){

first=null; second=null; lock=false;

turn=0;

scores=[0,0];

memory={};


board.innerHTML="";

deck=[...baseCards,...baseCards].sort(()=>Math.random()-0.5);


deck.forEach(c=>{

let card=document.createElement("div");

card.className="card";


let inner=document.createElement("div");

inner.className="inner";


let front=document.createElement("div");

front.className="face front";

front.innerHTML=`<img src="${getImg(c)}">`;


let back=document.createElement("div");

back.className="face back";

back.innerHTML=`<img src="${backImg}">`;


inner.append(front,back);

card.appendChild(inner);

card.dataset.value=c;


card.onclick=()=>onCardClick(card);


board.appendChild(card);

});


update();

}


function onCardClick(card){

if(turn!==0||lock||card.classList.contains("taken")||card===first) return;


flip(card);


if(!first){

first=card;

return;

}


second=card;

judge();

}


function update(){

document.getElementById("turn").textContent =

turn===0?"あなたのターン":"CPUのターン";

document.getElementById("score").textContent =

`あなた:${scores[0]} / CPU:${scores[1]}`;

}


function flip(card){

card.classList.add("flip");


memory[card.dataset.value]=memory[card.dataset.value]||[];

if(!memory[card.dataset.value].includes(card)){

memory[card.dataset.value].push(card);

}

}


function unflip(card){

card.classList.remove("flip");

}


function judge(){

lock=true;


setTimeout(()=>{

if(first.dataset.value===second.dataset.value){

first.classList.add("taken");

second.classList.add("taken");

scores[turn]++;

reset(true);

}else{

unflip(first);

unflip(second);

reset(false);

}

},500);

}


function reset(success){

first=null; second=null; lock=false;


if(!success) turn=1-turn;

update();


if(checkEnd()) return;


if(turn===1){

setTimeout(cpu,600);

}

}


/* CPU */

function cpu(){

let choices=[...document.querySelectorAll(".card:not(.flip):not(.taken)")];

let c1=choices[Math.floor(Math.random()*choices.length)];

let c2;

do{

c2=choices[Math.floor(Math.random()*choices.length)];

}while(c1===c2);


flip(c1);

first=c1;


setTimeout(()=>{

flip(c2);

second=c2;

judge();

},400);

}


/* 終了 */

function checkEnd(){

if(document.querySelectorAll(".taken").length===10){

setTimeout(showResult,300);

return true;

}

return false;

}


/* 結果 */

function showResult(){

let diff=document.getElementById("difficulty").value;


let text="";

if(scores[0]>scores[1]) text="あなたの勝ち！";

else if(scores[1]>scores[0]) text="CPUの勝ち！";

else text="引き分け！";


document.getElementById("resultText").textContent=text;


let key="best_"+diff;

let best=parseInt(localStorage.getItem(key)||"0");


if(scores[0]>best){

best=scores[0];

localStorage.setItem(key,best);

}


document.getElementById("highScore").textContent =

`${diff} ハイスコア: ${best}`;


document.getElementById("difficultyResult").value=diff;

document.getElementById("result").style.display="flex";


let amount = scores[0]>scores[1] ? 120 : 50;


for(let i=0;i<amount;i++){

let c=document.createElement("div");

c.className="confetti";

c.style.left=Math.random()*100+"%";

c.style.background=`hsl(${Math.random()*360},100%,50%)`;

c.style.animationDuration=(1+Math.random())+"s";

document.body.appendChild(c);

setTimeout(()=>c.remove(),2000);

}

}
